68use bevy::prelude::*;
69use wasm_bindgen::prelude::*;
70
71mod web_sys_mod;
72use web_sys_mod as wsm;
73mod state_dead_mod;
74mod state_in_game_mod;
75mod state_main_menu_mod;
76
77const CANVAS_WIDTH: i32 = 600;
78const CANVAS_HEIGHT: i32 = 600;
79const VERSION: &str = env!("CARGO_PKG_VERSION");
80
81#[derive(Clone, Copy, PartialEq, Eq, Hash, Debug, States)]
82#[states(scoped_entities)]
83enum AppState {
84    MainMenu,
85    InGame,
86    Dead,
87}
88
89#[wasm_bindgen]
90pub fn main() {
91    // rust has `Raw string literals` that are great!
92    // just add r# before the starting double quotes and # after the ending double quotes.
93    let html = format!(
94        r#"
95<canvas id="snake_bevy_canvas" width="{CANVAS_WIDTH}" height="{CANVAS_HEIGHT}"></canvas>
96"#
97    );
98
99    // WARNING for HTML INJECTION! Never put user provided strings in set_html_element_inner_html.
100    // Only correctly html encoded strings can use this function.
101    wsm::set_html_element_inner_html("div_for_wasm_html_injecting", &html);
102
103    // bevy initiation
104    let mut app = bevy::app::App::new();
105    app.add_plugins((
106        DefaultPlugins
107            .set(WindowPlugin {
108                primary_window: Some(Window {
109                    // provide the ID selector string here
110                    canvas: Some("#snake_bevy_canvas".into()),
111                    resolution: bevy::window::WindowResolution::new(CANVAS_WIDTH as f32, CANVAS_HEIGHT as f32), //.with_scale_factor_override(1.0),
112                    resizable: false,
113                    // ... any other window properties ...
114                    ..default()
115                }),
116                ..default()
117            })
118            .set(bevy::log::LogPlugin {
119                filter: "info,wgpu_core=error,wgpu_hal=error,bevy_render=error,bevy_ecs=error,bevy_winit=error,bevy_core_pipeline=error,bevy_pbr=error,symphonia=error,snake_bevy_wasm=debug".into(),
120                level: bevy::log::Level::DEBUG,
121                custom_layer: |_| None,
122            })
123            // don't download meta files for assets
124            .set(AssetPlugin {
125                meta_check: bevy::asset::AssetMetaCheck::Never,
126                ..default()
127            }),
128        bevy_kira_audio::AudioPlugin,
129    ));
130
131    info!("started snake_bevy_wasm {}", VERSION);
132
133    // initial state is MainMenu
134    app.insert_state(AppState::MainMenu);
135
136    state_main_menu_mod::add_main_menu_to_app(&mut app);
137
138    state_in_game_mod::add_in_game_to_app(&mut app);
139
140    state_dead_mod::add_dead_to_app(&mut app);
141
142    app.run();
143}
