28//!
29//! Everybody knows this game. This is an educational example.

31//! ## Graphics
32//!
33//! The canvas will be 1000x1000 pixels. The sprite will be 50x50 pixels. So the game-board will be 20x20 cells.
34//! The movements are based on elapsed time - tick.
35//! Bevy coordinate system is 0,0 in the middle of the canvas. Right is x+, Left is x-, Up is y+, Down in y-.
36//! For the game grid I will use the coordinate system 0,0 is in the up-left corner. Right is x+, Left is x-, Down in y+, Up is y-.
37//! Using this simplified coordinate system the array index is the same as the coordinate of the sprite.
38//! The game data will be in the game coordinate system. Then the renderer will transform that into the bevy coordinate system.
39//!  
40//!
41//! ## Faster builds
42//!
43//! I use the mold linker to speed up the build. In ~/.cargo/config.toml I already have:
44//!
45//! ```toml
46//! [target.x86_64-unknown-linux-gnu]
47//! rustflags = ["-C", "link-arg=-B/home/rustdevuser/.cargo/bin/mold"]
48//! ```
49//!
50//! But rust-analyzer does not read that file. In .vscode/settings.json I added
51//!
52//! ```json
53//!     "rust-analyzer.cargo.extraEnv": {
54//!         "RUSTFLAGS": "-Clink-arg=-B/home/rustdevuser/.cargo/bin/mold"
55//!     },
56//! ```
57//!
58//! Hopefully it will help but I don't really know for sure.
59//!
60//! The build step `wasm-opt` was really slow so, into Cargo.toml I added:
61//!
62//! ```toml
63//! [package.metadata.wasm-pack.profile.profiling]
64//! # for faster builds
65//! wasm-opt = false
66//! ```
67//!
68//! Now I can build faster in development using the mode `--profiling`
69//!
70//! ```bash
71//! wasm-pack build --target web --profiling
72//! ```
73//!
74//! For the release mode, wasm-opt was still too slow: 10 minutes.
75//! I copied the files from <https://github.com/WebAssembly/binaryen/releases/download/version_123/binaryen-version_123-x86_64-linux.tar.gz> from the bin/ folder into ~/bin. Then make them all executable `chmod +x *.*`. wasm-pack was able to find the executable wasm-opt in this folder.  
76//!
77//! Now I have wasm-opt version 123. This version should be faster.
78//! The wasm-opt now takes only 30 seconds! Much less than 10 minutes.
79//! Size reduction of the wasm file is from 65MB to 40MB.

136const CANVAS_WIDTH: i32 = 1000;
137const CANVAS_HEIGHT: i32 = 1000;
138const STEP_DURATION: f64 = 0.2;

196#[wasm_bindgen]
197pub fn main() {
198    // rust has `Raw string literals` that are great!
199    // just add r# before the starting double quotes and # after the ending double quotes.
200    let html = format!(
201        r#"
202<canvas id="snake_bevy_canvas" width="{CANVAS_WIDTH}" height="{CANVAS_HEIGHT}"></canvas>
203"#
204    );
205
206    // WARNING for HTML INJECTION! Never put user provided strings in set_html_element_inner_html.
207    // Only correctly html encoded strings can use this function.
208    wsm::set_html_element_inner_html("div_for_wasm_html_injecting", &html);
209
210    // bevy initiation
211    let mut app = bevy::app::App::new();
212    app.add_plugins(DefaultPlugins.set(WindowPlugin {
213        primary_window: Some(Window {
214            // provide the ID selector string here
215            canvas: Some("#snake_bevy_canvas".into()),
216            resolution: bevy::window::WindowResolution::new(CANVAS_WIDTH as f32, CANVAS_HEIGHT as f32).with_scale_factor_override(1.0),
217            resizable: false,
218            // ... any other window properties ...
219            ..default()
220        }),
221        ..default()
222    }));
223
224    // Set the Fixed Timestep interval for game logic to 0.x seconds
225    app.insert_resource(Time::<Fixed>::from_seconds(STEP_DURATION));
226    // only once on startup
227    app.add_systems(Startup, startup);
228    // game logic is sequential
229    app.add_systems(
230        FixedUpdate,
231        (snake_head_move, eat_bird.after(snake_head_move), move_segments.after(eat_bird), check_dead.after(move_segments)),
232    );
233    // render frame and react to events
234    app.add_systems(Update, (render_snake_head, render_bird, render_segment, handle_movement_input, render_dead, render_debug_text));
235    app.run();
236}

238// run on startup
239fn startup(mut commands: Commands, mut meshes: ResMut<Assets<Mesh>>, mut materials: ResMut<Assets<ColorMaterial>>) {
240    commands.spawn(Camera2d);
241    // snake head
242    let snake_head_position = Position { x: 10, y: 10 };
243    commands.spawn((
244        Mesh2d(meshes.add(Rectangle::new(50.0, 50.0))),
245        Transform::from_xyz(snake_head_position.to_bevy_x(), snake_head_position.to_bevy_y(), SNAKE_Z_LAYER),
246        MeshMaterial2d(materials.add(Color::hsl(300., 0.95, 0.7))),
247        SnakeHead {
248            position: snake_head_position.clone(),
249            direction: Direction::Down,
250            last_position: snake_head_position,
251            just_eating: false,
252            segment_len: 1,
253            updated: false,
254            dead: false,
255        },
256    ));
257
258    // first and only segment
259    let segment_position = Position { x: 10, y: 9 };
260    commands.spawn((
261        Mesh2d(meshes.add(Rectangle::new(50.0, 50.0))),
262        Transform::from_xyz(segment_position.to_bevy_x(), segment_position.to_bevy_y(), OTHER_Z_LAYER),
263        MeshMaterial2d(materials.add(Color::hsl(250., 0.95, 0.7))),
264        SnakeSegment {
265            position: segment_position,
266            index: 0,
267            updated: false,
268        },
269    ));
270
271    // spawn entity bird
272    let bird_position = Position { x: 9, y: 9 };
273    commands.spawn((
274        Mesh2d(meshes.add(Circle::new(25.0))),
275        MeshMaterial2d(materials.add(Color::hsl(2., 0.95, 0.7))),
276        Transform::from_xyz(bird_position.to_bevy_x(), bird_position.to_bevy_y(), OTHER_Z_LAYER),
277        Bird {
278            position: bird_position.clone(),
279            updated: false,
280        },
281    ));
282
283    // Text with one section
284    commands.spawn((
285        Visibility::Hidden,
286        // Accepts a `String` or any type that converts into a `String`, such as `&str`
287        Text::new("The snake\nis dead!"),
288        TextFont { font_size: 67.0, ..default() },
289        TextShadow::default(),
290        // Set the justification of the Text
291        TextLayout::new_with_justify(JustifyText::Center),
292        // Set the style of the Node itself.
293        Node {
294            position_type: PositionType::Absolute,
295            align_content: AlignContent::Center,
296            justify_content: JustifyContent::Center,
297            ..default()
298        },
299        AnimatedText,
300    ));
301
302    commands.spawn((
303        // Visibility::Hidden,
304        Text::new("Debug text: "),
305        Node {
306            position_type: PositionType::Absolute,
307            bottom: Val::Px(5.0),
308            left: Val::Px(15.0),
309            ..default()
310        },
311        DebugText {
312            bird_position: format!("{:?}", &bird_position),
313        },
314    ));
315}

317impl Position {
318    /// transform GameCoordinates to BevyCoordinates
319    pub fn to_bevy_x(&self) -> f32 {
320        self.x as f32 * SPRITE_WIDTH as f32 - (BOARD_CENTER as f32 * SPRITE_WIDTH as f32) + (SPRITE_WIDTH as f32 / 2.)
321    }
322    /// transform GameCoordinates to BevyCoordinates. Strange because it goes in the opposite direction.
323    pub fn to_bevy_y(&self) -> f32 {
324        -(self.y as f32) * SPRITE_HEIGHT as f32 + (BOARD_CENTER as f32 * SPRITE_HEIGHT as f32) - (SPRITE_HEIGHT as f32 / 2.)
325    }
