81#[derive(Clone, Copy, PartialEq, Eq, Hash, Debug, States)]
82#[states(scoped_entities)]
83enum AppState {

89#[derive(PartialEq)]
90pub enum Orientation {

92    Portrait,

95#[derive(Resource)]
96pub struct GameBoardCanvas {
97    client_width: i32,
98    client_height: i32,
99    orientation: Orientation,
100    board_canvas_width: i32,
101    board_canvas_height: i32,
102    sprite_width: f32,
103    sprite_height: f32,

106// This struct defines the data that will be passed to your shader
107#[derive(Asset, TypePath, AsBindGroup, Debug, Clone)]
108struct CustomMaterial {
109    #[uniform(0)]
110    color: LinearRgba,
111}
112
113/// The Material trait is very configurable, but comes with sensible defaults for all methods.
114/// You only need to implement functions for features that need non-default behavior. See the Material api docs for details!
115impl Material2d for CustomMaterial {
116    fn vertex_shader() -> ShaderRef {
117        "custom_material.wgsl".into()
118    }
119    fn fragment_shader() -> ShaderRef {
120        "custom_material.wgsl".into()
121    }
122}

124#[wasm_bindgen]
125pub fn main() {
126    // rust has `Raw string literals` that are great!
127    // just add r# before the starting double quotes and # after the ending double quotes.
128    let html = r#"
129<canvas id="snake_bevy_canvas" ></canvas>
130"#;
131
132    // WARNING for HTML INJECTION! Never put user provided strings in set_html_element_inner_html.
133    // Only correctly html encoded strings can use this function.
134    wsm::set_html_element_inner_html("div_for_wasm_html_injecting", html);
135
136    // bevy initiation
137    let mut app = bevy::app::App::new();
138    let game_board_canvas = get_game_board_canvas();
139
140    app.add_plugins((
141        DefaultPlugins
142            .set(WindowPlugin {
143                primary_window: Some(Window {
144                    // provide the ID selector string here
145                    canvas: Some("#snake_bevy_canvas".into()),
146                    // all client area
147                    resolution: bevy::window::WindowResolution::new(game_board_canvas.client_width as f32, game_board_canvas.client_height as f32),
148                    fit_canvas_to_parent: true,
149                    // ... any other window properties ...
150                    ..default()
151                }),
152                ..default()
153            })
154            .set(bevy::log::LogPlugin {
155                filter: "info,wgpu_core=error,wgpu_hal=error,bevy_render=error,bevy_ecs=error,bevy_winit=error,bevy_core_pipeline=error,bevy_pbr=error,symphonia=error,snake_bevy_wasm=debug".into(),
156                level: bevy::log::Level::DEBUG,
157                custom_layer: |_| None,
158            })
159            // don't download meta files for assets
160            .set(AssetPlugin {
161                meta_check: bevy::asset::AssetMetaCheck::Never,
162                ..default()
163            })
164            .set(ImagePlugin::default_nearest()),
165        bevy_kira_audio::AudioPlugin,
166        Material2dPlugin::<CustomMaterial>::default(),
167    ));
168
169    info!("started snake_bevy_wasm {}", VERSION);
170    // initial state is MainMenu
171    app.insert_state(AppState::MainMenu);
172
173    app.insert_resource(game_board_canvas);
174    state_main_menu_mod::add_main_menu_to_app(&mut app);
175
176    state_in_game_mod::add_in_game_to_app(&mut app);
177
178    state_dead_mod::add_dead_to_app(&mut app);
179
180    app.run();
181}

183// setup fit to window size on startup
184fn get_game_board_canvas() -> GameBoardCanvas {
185    // check viewport and define sizes
186    let client_width = wsm::get_client_width();
187    let client_height = wsm::get_client_height();
188
189    // landscape for PC monitor viewport is around 1280 x 712px
190    // portrait for mobile phone it is around 360px x 649px
191    // choose the smaller square to fit the window
192    let (game_square_width, orientation) = if client_width > client_height {
193        (client_height, Orientation::Landscape)
194    } else {
195        (client_width, Orientation::Portrait)
196    };
197    // return
198    GameBoardCanvas {
199        client_width,
200        client_height,
201        orientation,
202        board_canvas_width: game_square_width,
203        board_canvas_height: game_square_width,
204        sprite_width: game_square_width as f32 / BOARD_WIDTH as f32,
205        sprite_height: game_square_width as f32 / BOARD_HEIGHT as f32,
206    }
207}

209/// Bevy 0.16 is not setting the correct width and height for the canvas element.
210/// I don't know why. The css style is correct, but the attributes of the html element are wrong.
211/// I must check this and correct this size frequently.
212pub fn handle_browser_resize(mut game_board_canvas: ResMut<GameBoardCanvas>, mut window: bevy::ecs::system::Single<&mut bevy::window::Window, bevy::ecs::query::With<bevy::window::PrimaryWindow>>) {
213    let client_width = wsm::get_client_width();
214    let client_height = wsm::get_client_height();
215
216    if (window.resolution.width() as i32) != client_width || (window.resolution.height() as i32) != client_height {
217        // debug!("handle_browser_resize {client_width} {client_height}");
218        game_board_canvas.client_width = client_width;
219        game_board_canvas.client_height = client_height;
220        // landscape for PC monitor viewport is around 1280 x 712px
221        // portrait for mobile phone it is around 360px x 649px
222        // choose the smaller square to fit the window
223        let (game_square_width, orientation) = if client_width > client_height {
224            (client_height, Orientation::Landscape)
225        } else {
226            (client_width, Orientation::Portrait)
227        };
228        game_board_canvas.orientation = orientation;
229        game_board_canvas.board_canvas_width = game_square_width;
230        game_board_canvas.board_canvas_height = game_square_width;
231        game_board_canvas.sprite_width = game_square_width as f32 / BOARD_WIDTH as f32;
232        game_board_canvas.sprite_height = game_square_width as f32 / BOARD_HEIGHT as f32;
233
234        window.resolution.set(client_width as f32, client_height as f32);
235    }
236}

238/// Portrait: left-top will be the square canvas, bottom is the remaining
239pub fn portrait(game_board_canvas: &Res<'_, GameBoardCanvas>) -> Node {
240    Node {
241        // Use the CSS Grid algorithm for laying out this node
242        display: Display::Grid,
243        // Make node fill the entirety of its parent (in this case the window)
244        width: Val::Percent(100.0),
245        height: Val::Percent(100.0),
246        // Set the grid to have 2 rows
247        // flex(1) means the remaining space
248        grid_template_rows: vec![GridTrack::px(game_board_canvas.board_canvas_height as f32), GridTrack::flex(1.0)],
249        ..default()
250    }
251}
252
253/// Landscape: left-top will be the square canvas, left is the remaining
254pub fn landscape(game_board_canvas: &Res<'_, GameBoardCanvas>) -> Node {
255    Node {
256        // Use the CSS Grid algorithm for laying out this node
257        display: Display::Grid,
258        // Make node fill the entirety of its parent (in this case the window)
259        width: Val::Percent(100.0),
260        height: Val::Percent(100.0),
261        // Set the grid to have 2 columns
262        // flex(1) means the remaining space
263        grid_template_columns: vec![GridTrack::px(game_board_canvas.board_canvas_width as f32), GridTrack::flex(1.0)],
264        ..default()
265    }
