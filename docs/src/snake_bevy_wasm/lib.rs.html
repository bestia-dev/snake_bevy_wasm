63use bevy::prelude::*;
64use wasm_bindgen::prelude::*;
65
66mod web_sys_mod;
67use web_sys_mod as wsm;
68mod state_dead_mod;
69mod state_in_game_mod;
70mod state_main_menu_mod;
71
72const CANVAS_WIDTH: i32 = 512;
73const CANVAS_HEIGHT: i32 = 512;
74const VERSION: &str = env!("CARGO_PKG_VERSION");
75
76#[derive(Clone, Copy, PartialEq, Eq, Hash, Debug, States)]
77#[states(scoped_entities)]
78enum AppState {
79    MainMenu,
80    InGame,
81    Dead,
82}
83
84#[wasm_bindgen]
85pub fn main() {
86    // rust has `Raw string literals` that are great!
87    // just add r# before the starting double quotes and # after the ending double quotes.
88    let html = format!(
89        r#"
90<canvas id="snake_bevy_canvas" width="{CANVAS_WIDTH}" height="{CANVAS_HEIGHT}"></canvas>
91"#
92    );
93
94    // WARNING for HTML INJECTION! Never put user provided strings in set_html_element_inner_html.
95    // Only correctly html encoded strings can use this function.
96    wsm::set_html_element_inner_html("div_for_wasm_html_injecting", &html);
97
98    // bevy initiation
99    let mut app = bevy::app::App::new();
100    app.add_plugins(
101        DefaultPlugins
102            .set(WindowPlugin {
103                primary_window: Some(Window {
104                    // provide the ID selector string here
105                    canvas: Some("#snake_bevy_canvas".into()),
106                    resolution: bevy::window::WindowResolution::new(CANVAS_WIDTH as f32, CANVAS_HEIGHT as f32), //.with_scale_factor_override(1.0),
107                    resizable: false,
108                    // ... any other window properties ...
109                    ..default()
110                }),
111                ..default()
112            })
113            .set(bevy::log::LogPlugin {
114                filter: "info,wgpu_core=error,wgpu_hal=error,bevy_render=error,bevy_ecs=error,bevy_winit=error,bevy_core_pipeline=error,bevy_pbr=error,snake_bevy_wasm=debug".into(),
115                level: bevy::log::Level::DEBUG,
116                custom_layer: |_| None,
117            }),
118    );
119
120    info!("snake_bevy_wasm {}", VERSION);
121
122    // initial state is MainMenu
123    app.insert_state(AppState::MainMenu);
124
125    state_main_menu_mod::add_main_menu_to_app(&mut app);
126
127    state_in_game_mod::add_in_game_to_app(&mut app);
128
129    state_dead_mod::add_dead_to_app(&mut app);
130
131    app.run();
132}
