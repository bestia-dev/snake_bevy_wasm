102pub fn add_in_game_to_app(app: &mut App) {
103    // Set the Fixed Timestep interval for game logic to 0.x seconds
104    app.insert_resource(Time::<Fixed>::from_seconds(STEP_DURATION));
105    app.add_systems(OnEnter(AppState::InGame), on_enter_in_game);
106    app.add_systems(OnExit(AppState::InGame), on_exit_in_game);
107    app.add_systems(
108        FixedUpdate,
109        (
110            move_snake_head.run_if(in_state(AppState::InGame)),
111            eat_bird.run_if(in_state(AppState::InGame)),
112            move_segments.run_if(in_state(AppState::InGame)),
113            check_dead.run_if(in_state(AppState::InGame)),
114            render_snake_head.run_if(in_state(AppState::InGame)),
115            render_segment.run_if(in_state(AppState::InGame)),
116        )
117            .chain(),
118    );
119
120    app.add_systems(
121        // render frame and react to events
122        Update,
123        (
124            crate::handle_browser_resize.run_if(in_state(AppState::InGame)),
125            button_interaction_system.run_if(in_state(AppState::InGame)),
126            // draw_axis.run_if(in_state(AppState::InGame)),
127            render_bird.run_if(in_state(AppState::InGame)),
128            handle_movement_input.run_if(in_state(AppState::InGame)),
129            render_points_text.run_if(in_state(AppState::InGame)),
130            render_debug_text.run_if(in_state(AppState::InGame)),
131        ),
132    );
133}

136fn on_enter_in_game(mut commands: Commands, asset_server: Res<AssetServer>, audio: Res<bevy_kira_audio::Audio>, game_board_canvas: Res<GameBoardCanvas>) {
137    let handle = audio.play(asset_server.load("snake_hiss.mp3")).looped().handle();
138    commands.insert_resource(InstanceHandle(handle));
139
140    commands.spawn(Camera2d);
141
142    let mut client = if game_board_canvas.orientation == Orientation::Landscape {
143        commands.spawn(crate::landscape(&game_board_canvas))
144    } else {
145        commands.spawn(crate::portrait(&game_board_canvas))
146    };
147
148    client.with_children(|client| {
149        client.spawn((
150            Node {
151                // Make node fill the entirety of its parent (in this case the window)
152                display: Display::Grid,
153                width: Val::Percent(100.0),
154                height: Val::Percent(100.0),
155                grid_template_rows: vec![GridTrack::percent(33.), GridTrack::percent(33.), GridTrack::percent(34.)],
156                ..default()
157            },
158            Outline {
159                width: Val::Px(2.),
160                offset: Val::Px(-1.),
161                color: Color::BLACK,
162            },
163        ));
164        // second cell
165        if game_board_canvas.orientation == Orientation::Portrait {
166            let mut keys = client.spawn((
167                // the UI buttons will despawn, but the rest of the game
168                // will remain in the background after death
169                StateScoped(AppState::InGame),
170                Node {
171                    // Use the CSS Grid algorithm for laying out this node
172                    display: Display::Grid,
173                    // Make node fill the entirety of its parent (in this case the window)
174                    width: Val::Percent(100.0),
175                    height: Val::Percent(100.0),
176                    align_items: AlignItems::Center,
177                    justify_content: JustifyContent::Center,
178                    // grid 5 col x 4 rows
179                    grid_template_columns: vec![
180                        GridTrack::percent(20.),
181                        GridTrack::percent(20.),
182                        GridTrack::percent(20.),
183                        GridTrack::percent(20.),
184                        GridTrack::percent(20.),
185                    ],
186                    grid_template_rows: vec![GridTrack::percent(25.), GridTrack::percent(25.), GridTrack::percent(25.), GridTrack::percent(25.)],
187                    ..default()
188                },
189                Outline {
190                    width: Val::Px(1.),
191                    offset: Val::Px(-2.),
192                    color: Color::from(RED),
193                },
194            ));
195
196            keys.with_children(|keys| {
197                keys.spawn((Node {
198                    display: Display::Grid,
199                    // Make this node span 5 grid columns so that it takes up the entire top row
200                    grid_column: GridPlacement::span(5),
201                    ..default()
202                },));
203                keys.spawn((Node {
204                    display: Display::Grid,
205                    // Make this node span 1 grid columns so that it takes up half row
206                    grid_column: GridPlacement::span(2),
207                    ..default()
208                },));
209                keys.spawn((
210                    Button,
211                    ButtonEnum::KeyUp,
212                    ImageNode {
213                        image: asset_server.load("key_up.png"),
214                        ..default()
215                    },
216                    Interaction::None,
217                    Node {
218                        width: Val::Px(80.),
219                        height: Val::Px(80.),
220                        // horizontally center child text
221                        justify_content: JustifyContent::Center,
222                        // vertically center child text
223                        align_items: AlignItems::Center,
224                        ..default()
225                    },
226                ));
227
228                keys.spawn((Node {
229                    display: Display::Grid,
230                    // Make this node span 1 grid columns so that it takes up the second half row
231                    grid_column: GridPlacement::span(2),
232                    ..default()
233                },));
234
235                keys.spawn((Node {
236                    display: Display::Grid,
237                    // Make this node span 1 grid columns so that it takes up one column
238                    grid_column: GridPlacement::span(1),
239                    ..default()
240                },));

242                keys.spawn((
243                    Button,
244                    ButtonEnum::KeyLeft,

246                        image: asset_server.load("key_left.png"),

254                        justify_content: JustifyContent::Center,
255                        // vertically center child text
256                        align_items: AlignItems::Center,
257                        ..default()
258                    },

261                keys.spawn((
262                    Button,
263                    ButtonEnum::KeyDown,
264                    ImageNode {
265                        image: asset_server.load("key_down.png"),
266                        ..default()
267                    },
268                    Interaction::None,
269                    Node {
270                        width: Val::Px(80.),
271                        height: Val::Px(80.),
272                        // horizontally center child text
273                        justify_content: JustifyContent::Center,

281                    Button,
282                    ButtonEnum::KeyRight,
283                    ImageNode {
284                        image: asset_server.load("key_right.png"),

302    // snake head
303    let snake_head_position = Position { x: 10, y: 10 };
304    commands.spawn((
305        Sprite::from_image(asset_server.load("snake_head_left.png")),
306        Transform::from_xyz(snake_head_position.to_bevy_x(&game_board_canvas), snake_head_position.to_bevy_y(&game_board_canvas), SNAKE_Z_LAYER).with_rotation(Quat::from_rotation_z(PI * 0.5)),

311            last_direction: Direction::Down,
312            last_position: snake_head_position,

314            moves: 0,

320    let segment_position = Position { x: 10, y: 9 };
321    commands.spawn((
322        Sprite::from_image(asset_server.load("segment_horizontal.png")),
323        Transform::from_xyz(segment_position.to_bevy_x(&game_board_canvas), segment_position.to_bevy_y(&game_board_canvas), OTHER_Z_LAYER).with_rotation(Quat::from_rotation_z(PI * 0.5)),
324        SnakeSegment {
325            position: segment_position,
326            direction: Direction::Down,
327            last_direction: Direction::Down,
328            is_tail: false,
329        },
330        SnakeSegmentIndex { index: 0 },
331    ));
332
333    // last (1) segment
334    let segment_position = Position { x: 10, y: 8 };
335    commands.spawn((
336        Sprite::from_image(asset_server.load("segment_tail.png")),
337        Transform::from_xyz(segment_position.to_bevy_x(&game_board_canvas), segment_position.to_bevy_y(&game_board_canvas), OTHER_Z_LAYER).with_rotation(Quat::from_rotation_z(PI * 0.5)),
338        SnakeSegment {
339            position: segment_position,
340            direction: Direction::Down,
341            last_direction: Direction::Down,
342            is_tail: true,
343        },
344        SnakeSegmentIndex { index: 1 },
345    ));
346
347    // spawn entity bird
348    let bird_position = Position { x: 9, y: 9 };
349    commands.spawn((
350        Sprite::from_image(asset_server.load("bird.png")),
351        Transform::from_xyz(bird_position.to_bevy_x(&game_board_canvas), bird_position.to_bevy_y(&game_board_canvas), BIRD_Z_LAYER),
352        Bird {
353            position: bird_position.clone(),
354            color: WHITE.into(),
355        },
356    ));
357
358    commands.spawn((
359        Text::new(""),
360        Node {
361            position_type: PositionType::Absolute,
362            bottom: Val::Px(5.0),
363            left: Val::Px(15.0),
364            ..default()
365        },
366        PointsText {},
367    ));
368
369    // DebugText
370    commands.spawn((
371        Visibility::Hidden,
372        Text::new("Debug text: "),
373        Node {
374            position_type: PositionType::Absolute,
375            bottom: Val::Px(25.0),
376            left: Val::Px(300.0),
377            ..default()
378        },
379        DebugText {
380            bird_position: format!("{:?}", &bird_position),
381        },
382    ));
383}

385fn on_exit_in_game(handle: Res<InstanceHandle>, mut audio_instances: ResMut<Assets<AudioInstance>>) {
386    // stop only the music, not the other sound effects
387    if let Some(instance) = audio_instances.get_mut(&handle.0) {
388        instance.stop(AudioTween::default());
389    }
390}

392impl Position {
393    /// transform GameCoordinates to BevyCoordinates
394    pub fn to_bevy_x(&self, game_board_canvas: &GameBoardCanvas) -> f32 {
395        self.x as f32 * game_board_canvas.sprite_width - (game_board_canvas.client_width as f32 / 2.0) + (game_board_canvas.sprite_width / 2.)
396    }
397    /// transform GameCoordinates to BevyCoordinates. Strange because it goes in the opposite direction.
398    pub fn to_bevy_y(&self, game_board_canvas: &GameBoardCanvas) -> f32 {
399        -(self.y as f32) * game_board_canvas.sprite_height + (game_board_canvas.client_height as f32 / 2.0) - (game_board_canvas.sprite_height / 2.)
400    }
