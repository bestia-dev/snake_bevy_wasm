9pub fn add_main_menu_to_app(app: &mut App) {
10    app.add_systems(OnEnter(AppState::MainMenu), on_enter_main_menu);
11
12    // MUST add all systems to app with run_if in_state MainMenu
13    app.add_systems(Update, handle_main_menu_ui_input.run_if(in_state(AppState::MainMenu)));
14}

16pub fn on_enter_main_menu(mut commands: Commands) {
17    commands.spawn(Camera2d);
18    // Text with one section
19    let mut grid = commands.spawn((
20        StateScoped(AppState::MainMenu),
21        Node {
22            // Use the CSS Grid algorithm for laying out this node
23            display: Display::Grid,
24            // Make node fill the entirety of its parent (in this case the window)
25            width: Val::Percent(100.0),
26            height: Val::Percent(100.0),
27            // Set the grid to have 3 rows with sizes [20px, auto, 20px]
28            grid_template_rows: vec![GridTrack::vw(33.), GridTrack::vw(33.), GridTrack::vw(33.)],
29            ..default()
30        },
31    ));
32    {
33        grid.with_children(|grid| {
34            // Header
35            let mut header_box = grid.spawn((Node {
36                flex_direction: FlexDirection::Column,
37                align_items: AlignItems::Center,
38                justify_content: JustifyContent::Center,
39
40                ..default()
41            },));
42            {
43                header_box.with_children(|header_box| {
44                    // Header
45                    header_box.spawn((
46                        Text::new("bestia.dev/snake_bevy_wasm"),
47                        TextFont {
48                            font_size: SPRITE_HEIGHT as f32,
49                            ..default()
50                        },
51                        TextLayout::new_with_justify(JustifyText::Center),
52                        TextColor::from(GREEN),
53                    ));
54                });
55            }
56            // middle
57            let mut middle_box = grid.spawn((Node {
58                flex_direction: FlexDirection::Column,
59                align_items: AlignItems::Center,
60                justify_content: JustifyContent::Center,
61                ..default()
62            },));
63            {
64                middle_box.with_children(|middle_box| {
65                    // middle
66                    middle_box.spawn((
67                        Text::new(format!("Bestia.dev tutorial\nRust+Bevy+Wasm v{VERSION}")),
68                        TextFont {
69                            font_size: SPRITE_HEIGHT as f32,
70                            ..default()
71                        },
72                        TextLayout::new_with_justify(JustifyText::Center),
73                        TextColor::from(YELLOW),
74                    ));
75                });
76            }
77            // footer
78            let mut footer_box = grid.spawn((Node {
79                flex_direction: FlexDirection::Column,
80                align_items: AlignItems::Center,
81                justify_content: JustifyContent::Center,
82                ..default()
83            },));
84            {
85                footer_box.with_children(|footer_box| {
86                    // footer
87                    footer_box.spawn((
88                        Text::new("Press N to start"),
89                        TextFont {
90                            font_size: SPRITE_HEIGHT as f32,
91                            ..default()
92                        },
93                        TextLayout::new_with_justify(JustifyText::Center),
94                        TextColor::from(RED),
95                    ));
96                });
97            }
98        });
99    }
100}

102pub fn handle_main_menu_ui_input(keys: Res<ButtonInput<KeyCode>>, mut next_state: ResMut<NextState<AppState>>) {
103    if keys.pressed(KeyCode::KeyN) {
104        next_state.set(AppState::InGame);
