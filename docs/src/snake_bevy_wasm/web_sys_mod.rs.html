17/// return the global window object
18pub fn window() -> web_sys::Window {
19    unwrap!(web_sys::window())
20}

22/// get element by id
23pub fn get_element_by_id(element_id: &str) -> web_sys::Element {
24    let document = unwrap!(window().document());
25    unwrap!(document.get_element_by_id(element_id))
26}

34/// get html element by id
35pub fn get_html_element_by_id(element_id: &str) -> web_sys::HtmlElement {
36    let element = get_element_by_id(element_id);
37    let html_element: web_sys::HtmlElement = unwrap!(element.dyn_into::<web_sys::HtmlElement>());
38    //return
39    html_element
40}

42/// get input element value string by id
43pub fn get_input_element_value_string_by_id(element_id: &str) -> String {
44    let input_element = get_element_by_id(element_id);
45    let input_html_element = unwrap!(input_element.dyn_into::<web_sys::HtmlInputElement>());
46    input_html_element.value()
47}

49/// add event listener for button
50pub fn add_listener_to_button(element_id: &str, fn_on_click_button: &'static (dyn Fn() + 'static)) {
51    let handler_1 = Box::new(move || {
52        fn_on_click_button();
53    }) as Box<dyn FnMut()>;
54    let closure = Closure::wrap(handler_1);
55
56    let html_element = get_html_element_by_id(element_id);
57    html_element.set_onclick(Some(closure.as_ref().unchecked_ref()));
58    closure.forget();
59}
