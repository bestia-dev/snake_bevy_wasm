11// fixed time every 0.5 seconds
12pub fn move_snake_head(mut snake_query: Query<&mut SnakeHead>) {
13    if let Ok(mut snake_head) = snake_query.single_mut() {
14        snake_head.last_direction = snake_head.direction.clone();
15        snake_head.direction = snake_head.new_direction.clone();
16
17        snake_head.last_position = snake_head.position.clone();
18
19        match &snake_head.direction {
20            Direction::Up => snake_head.position.y -= 1,
21            Direction::Down => snake_head.position.y += 1,
22            Direction::Left => snake_head.position.x -= 1,
23            Direction::Right => snake_head.position.x += 1,
24        }
25        snake_head.updated = true;
26        snake_head.moves += 1;
27    }
28}

30// this is executed after snake_head_move
31pub fn check_dead(
32    mut snake_query: Query<&mut SnakeHead>,
33    segment_query: Query<&mut SnakeSegment>,
34    mut next_state: ResMut<NextState<AppState>>,
35    asset_server: Res<AssetServer>,
36    audio: Res<bevy_kira_audio::Audio>,
37) {
38    if let Ok(snake_head) = snake_query.single_mut() {
39        if snake_head.position.x < 0 || snake_head.position.y < 0 || snake_head.position.x >= BOARD_WIDTH || snake_head.position.y >= BOARD_HEIGHT {
40            next_state.set(AppState::Dead);
41            audio.play(asset_server.load("game_over.mp3"));
42        }
43        for segment in segment_query {
44            if segment.position == snake_head.position {
45                next_state.set(AppState::Dead);
46                audio.play(asset_server.load("game_over.mp3"));
47                break;
48            }
49        }
50    }
51}

53pub fn eat_bird(
54    _time: Res<Time>,
55    mut snake_query: Query<&mut SnakeHead>,
56    mut bird_query: Query<&mut Bird>,
57    mut debug_text_query: Query<&mut DebugText>,
58    asset_server: Res<AssetServer>,
59    audio: Res<bevy_kira_audio::Audio>,
60) {
61    if let Ok(mut snake_head) = snake_query.single_mut() {
62        if let Ok(mut bird) = bird_query.single_mut() {
63            if snake_head.position == bird.position {
64                audio.play(asset_server.load("bird_chirp.mp3"));
65
66                snake_head.just_eating = true;
67                snake_head.points += 1;
68                // food: point, longer body
69                // new random position
70                bird.position.x = ops::floor(js_sys::Math::random() as f32 * BOARD_WIDTH as f32) as i32;
71                bird.position.y = ops::floor(js_sys::Math::random() as f32 * BOARD_HEIGHT as f32) as i32;
72                bird.updated = true;
73                if let Ok(mut debug_text) = debug_text_query.single_mut() {
74                    debug_text.bird_position = format!("{:?}", &bird.position);
75                }
76            }
77        }
78    }
79}

81/// first segment is after the snake head
82pub fn move_segments(mut commands: Commands, mut snake_query: Query<&mut SnakeHead>, mut segment_query: Query<&mut SnakeSegment>, asset_server: Res<AssetServer>) {
83    if let Ok(mut snake_head) = snake_query.single_mut() {
84        let segment_len = segment_query.iter().len();
85
86        // Sort according to `usize index`.
87        let mut sorted_snake_segments: Vec<_> = segment_query.iter_mut().sort_by::<&SnakeSegment>(|value_1, value_2| value_1.index.cmp(&value_2.index)).collect();
88
89        for snake_segment in sorted_snake_segments.iter_mut() {
90            snake_segment.index += 1;
91        }
92
93        // the last segment becomes the first (zero) to avoid new spawn()
94        let new_first_segment = sorted_snake_segments.last_mut().unwrap();
95        // clone the old values, they will be used to make the tail longer if eating
96        let last_segment_clone = new_first_segment.clone();
97
98        new_first_segment.index = 0;
99        new_first_segment.position = snake_head.last_position.clone();
100        new_first_segment.direction = snake_head.direction.clone();
101        new_first_segment.last_direction = snake_head.last_direction.clone();
102
103        new_first_segment.updated = true;
104
105        // I will use the last_segment_clone to spawn the new segment, index is segment_len
106        if snake_head.just_eating {
107            snake_head.just_eating = false;
108
109            let rotation = match last_segment_clone.direction {
110                Direction::Up => Quat::from_rotation_z(-PI * 0.5),
111                Direction::Right => Quat::from_rotation_z(PI),
112                Direction::Down => Quat::from_rotation_z(PI * 0.5),
113                Direction::Left => Quat::from_rotation_z(0.),
114            };
115            commands.spawn((
116                StateScoped(AppState::InGame),
117                Sprite::from_image(asset_server.load("segment_tail.png")),
118                Transform::from_xyz(last_segment_clone.position.to_bevy_x(), last_segment_clone.position.to_bevy_y(), OTHER_Z_LAYER).with_rotation(rotation),
119                SnakeSegment {
120                    position: last_segment_clone.position.clone(),
121                    index: segment_len,
122                    direction: last_segment_clone.direction.clone(),
123                    last_direction: last_segment_clone.last_direction.clone(),
124                    updated: false,
125                },
126            ));
127        }
128    }
