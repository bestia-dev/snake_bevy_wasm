8pub fn render_snake_head(mut snake_head_query: Query<(&mut SnakeHead, &mut Transform), Changed<SnakeHead>>) {
9    if let Ok((snake_head, mut transform)) = snake_head_query.single_mut() {
10        transform.translation.x = snake_head.position.to_bevy_x();
11        transform.translation.y = snake_head.position.to_bevy_y();
12
13        // TODO: hot to flip using look_at???
14        // rotate and/or flip head
15        match snake_head.direction {
16            Direction::Up => {
17                transform.look_at(
18                    Vec3 {
19                        x: snake_head.position.to_bevy_x(),
20                        y: snake_head.position.to_bevy_y(),
21                        z: f32::INFINITY,
22                    },
23                    Vec3::X,
24                );
25            }
26            Direction::Right => {
27                transform.look_at(
28                    Vec3 {
29                        x: snake_head.position.to_bevy_x(),
30                        y: snake_head.position.to_bevy_y(),
31                        z: f32::INFINITY,
32                    },
33                    Vec3 { x: 0.0, y: -1.0, z: 0.0 },
34                );
35                transform.rotate_x(PI);
36            }
37            Direction::Down => {
38                transform.look_at(
39                    Vec3 {
40                        x: snake_head.position.to_bevy_x(),
41                        y: snake_head.position.to_bevy_y(),
42                        z: f32::INFINITY,
43                    },
44                    -Vec3::X,
45                );
46            }
47            Direction::Left => {
48                transform.look_at(
49                    Vec3 {
50                        x: snake_head.position.to_bevy_x(),
51                        y: snake_head.position.to_bevy_y(),
52                        z: f32::INFINITY,
53                    },
54                    Vec3::Y,
55                );
56            }
57        };
58    }
59}

68pub fn render_segment(mut segment_query: Query<(&mut SnakeSegment, &mut Transform, &mut Sprite), Or<(Changed<SnakeSegment>, Added<SnakeSegment>)>>, asset_server: Res<AssetServer>) {
69    for (segment, mut transform, mut sprite) in segment_query.iter_mut() {
70        transform.translation.x = segment.position.to_bevy_x();
71        transform.translation.y = segment.position.to_bevy_y();
72
73        // the last segment is the tail
74        if segment.is_tail {
75            sprite.image = asset_server.load("segment_tail.png");
76        } else if segment.direction != segment.last_direction {
77            sprite.image = asset_server.load("segment_corner.png");
78        } else {
79            sprite.image = asset_server.load("segment_horizontal.png");
80        }
81        rotate_segment(&segment, transform);
82    }
83}

85// corner can be clockwise and counterclockwise
86fn rotate_segment(segment: &Mut<'_, SnakeSegment>, mut transform: Mut<'_, Transform>) {
87    let forward = Vec3 {
88        x: segment.position.to_bevy_x(),
89        y: segment.position.to_bevy_y(),
90        z: f32::INFINITY,
91    };
92
93    match segment.direction {
94        Direction::Up => {
95            transform.look_at(forward, Vec3::X);
96            // if anti-clockwise
97            if segment.last_direction == Direction::Right {
98                transform.rotate_y(PI);
99            }
100        }
101        Direction::Right => {
102            transform.look_at(forward, -Vec3::Y);
103            // if anti-clockwise
104            if segment.last_direction == Direction::Down {
105                transform.rotate_x(PI);
106            }
107        }
108        Direction::Down => {
109            transform.look_at(forward, -Vec3::X);
110            // if anti-clockwise
111            if segment.last_direction == Direction::Left {
112                transform.rotate_y(PI);
113            }
114        }
115        Direction::Left => {
116            transform.look_at(forward, Vec3::Y);
117            // if anti-clockwise
118            if segment.last_direction == Direction::Up {
119                transform.rotate_x(PI);
120            }
121        }
122    }
123}
