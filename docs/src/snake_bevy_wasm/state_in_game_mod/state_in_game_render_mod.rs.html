19pub fn render_snake_head(mut snake_head_query: Query<(&mut SnakeHead, &mut Transform), Changed<SnakeHead>>, game_board_canvas: Res<GameBoardCanvas>) {
20    if let Ok((snake_head, mut transform)) = snake_head_query.single_mut() {
21        transform.translation.x = snake_head.position.to_bevy_x(&game_board_canvas);
22        transform.translation.y = snake_head.position.to_bevy_y(&game_board_canvas);
23
24        // TODO: hot to flip using look_at???
25        // rotate and/or flip head
26        match snake_head.direction {
27            Direction::Up => {
28                transform.look_at(
29                    Vec3 {
30                        x: snake_head.position.to_bevy_x(&game_board_canvas),
31                        y: snake_head.position.to_bevy_y(&game_board_canvas),
32                        z: f32::INFINITY,
33                    },
34                    Vec3::X,
35                );
36            }
37            Direction::Right => {
38                transform.look_at(
39                    Vec3 {
40                        x: snake_head.position.to_bevy_x(&game_board_canvas),
41                        y: snake_head.position.to_bevy_y(&game_board_canvas),
42                        z: f32::INFINITY,
43                    },
44                    Vec3 { x: 0.0, y: -1.0, z: 0.0 },
45                );
46                transform.rotate_x(PI);
47            }
48            Direction::Down => {
49                transform.look_at(
50                    Vec3 {
51                        x: snake_head.position.to_bevy_x(&game_board_canvas),
52                        y: snake_head.position.to_bevy_y(&game_board_canvas),
53                        z: f32::INFINITY,
54                    },
55                    -Vec3::X,
56                );
57            }
58            Direction::Left => {
59                transform.look_at(
60                    Vec3 {
61                        x: snake_head.position.to_bevy_x(&game_board_canvas),
62                        y: snake_head.position.to_bevy_y(&game_board_canvas),
63                        z: f32::INFINITY,
64                    },
65                    Vec3::Y,
66                );
67            }
68        };
69    }
70}
71pub fn render_bird(mut queried_entities: Query<(&mut Bird, &mut Transform, &mut Sprite), Changed<Bird>>, game_board_canvas: Res<GameBoardCanvas>) {
72    if let Ok((bird, mut transform, mut sprite)) = queried_entities.single_mut() {
73        transform.translation.x = bird.position.to_bevy_x(&game_board_canvas);
74        transform.translation.y = bird.position.to_bevy_y(&game_board_canvas);

80pub fn render_segment(
81    mut segment_query: Query<(&mut SnakeSegment, &mut Transform, &mut Sprite), Or<(Changed<SnakeSegment>, Added<SnakeSegment>)>>,
82    asset_server: Res<AssetServer>,
83    game_board_canvas: Res<GameBoardCanvas>,
84) {
85    for (segment, mut transform, mut sprite) in segment_query.iter_mut() {
86        transform.translation.x = segment.position.to_bevy_x(&game_board_canvas);
87        transform.translation.y = segment.position.to_bevy_y(&game_board_canvas);
88
89        // the last segment is the tail
90        if segment.is_tail {
91            sprite.image = asset_server.load("segment_tail.png");
92        } else if segment.direction != segment.last_direction {
93            sprite.image = asset_server.load("segment_corner.png");
94        } else {
95            sprite.image = asset_server.load("segment_horizontal.png");
96        }
97        rotate_segment(&segment, transform, &game_board_canvas);
98    }
99}

101// corner can be clockwise and counterclockwise
102fn rotate_segment(segment: &Mut<'_, SnakeSegment>, mut transform: Mut<'_, Transform>, game_board_canvas: &Res<GameBoardCanvas>) {
103    let forward = Vec3 {
104        x: segment.position.to_bevy_x(game_board_canvas),
105        y: segment.position.to_bevy_y(game_board_canvas),
106        z: f32::INFINITY,
107    };
108
109    match segment.direction {
110        Direction::Up => {
111            transform.look_at(forward, Vec3::X);
112            // if anti-clockwise
113            if segment.last_direction == Direction::Right {
114                transform.rotate_y(PI);
115            }
116        }
117        Direction::Right => {
118            transform.look_at(forward, -Vec3::Y);
119            // if anti-clockwise
120            if segment.last_direction == Direction::Down {
121                transform.rotate_x(PI);
122            }
123        }
124        Direction::Down => {
125            transform.look_at(forward, -Vec3::X);
126            // if anti-clockwise
127            if segment.last_direction == Direction::Left {
128                transform.rotate_y(PI);
129            }
130        }
131        Direction::Left => {
132            transform.look_at(forward, Vec3::Y);
133            // if anti-clockwise
134            if segment.last_direction == Direction::Up {
135                transform.rotate_x(PI);
136            }
137        }
138    }
139}
