8pub fn handle_movement_input(keys: Res<ButtonInput<KeyCode>>, mut snake_head: Single<&mut SnakeHead>, mut next_state: ResMut<NextState<AppState>>) {
9    if (keys.pressed(KeyCode::ArrowUp) || keys.pressed(KeyCode::KeyW)) && snake_head.direction != Direction::Down {
10        snake_head.new_direction = Direction::Up;
11    } else if (keys.pressed(KeyCode::ArrowRight) || keys.pressed(KeyCode::KeyD)) && snake_head.direction != Direction::Left {
12        snake_head.new_direction = Direction::Right;
13    } else if (keys.pressed(KeyCode::ArrowDown) || keys.pressed(KeyCode::KeyS)) && snake_head.direction != Direction::Up {
14        snake_head.new_direction = Direction::Down;
15    } else if (keys.pressed(KeyCode::ArrowLeft) || keys.pressed(KeyCode::KeyA)) && snake_head.direction != Direction::Right {
16        snake_head.new_direction = Direction::Left;
17    } else if keys.pressed(KeyCode::KeyX) {
18        next_state.set(AppState::MainMenu);
19    }
20}

22#[allow(clippy::type_complexity)]
23pub fn button_interaction_system(
24    mut snake_head: Single<&mut SnakeHead>,
25    interaction_query: Query<(&ButtonEnum, &Interaction), (Changed<Interaction>, With<Button>)>,
26    mut next_state: ResMut<NextState<AppState>>,
27) {
28    for interaction in interaction_query {
29        if *interaction.1 == Interaction::Pressed {
30            if *interaction.0 == ButtonEnum::KeyUp {
31                snake_head.new_direction = Direction::Up;
32            } else if *interaction.0 == ButtonEnum::KeyRight {
33                snake_head.new_direction = Direction::Right;
34            } else if *interaction.0 == ButtonEnum::KeyDown {
35                snake_head.new_direction = Direction::Down;
36            } else if *interaction.0 == ButtonEnum::KeyLeft {
37                snake_head.new_direction = Direction::Left;
38            } else if *interaction.0 == ButtonEnum::KeyX {
39                next_state.set(AppState::MainMenu);
40            }
41        }
42    }
