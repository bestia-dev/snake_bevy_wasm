8pub fn handle_movement_input(keys: Res<ButtonInput<KeyCode>>, mut queried_entities: Query<&mut SnakeHead>, mut next_state: ResMut<NextState<AppState>>) {
9    if let Ok(mut snake_head) = queried_entities.single_mut() {
10        if (keys.pressed(KeyCode::ArrowUp) || keys.pressed(KeyCode::KeyW)) && snake_head.direction != Direction::Down {
11            snake_head.new_direction = Direction::Up;
12        } else if (keys.pressed(KeyCode::ArrowRight) || keys.pressed(KeyCode::KeyD)) && snake_head.direction != Direction::Left {
13            snake_head.new_direction = Direction::Right;
14        } else if (keys.pressed(KeyCode::ArrowDown) || keys.pressed(KeyCode::KeyS)) && snake_head.direction != Direction::Up {
15            snake_head.new_direction = Direction::Down;
16        } else if (keys.pressed(KeyCode::ArrowLeft) || keys.pressed(KeyCode::KeyA)) && snake_head.direction != Direction::Right {
17            snake_head.new_direction = Direction::Left;
18        } else if keys.pressed(KeyCode::KeyX) {
19            next_state.set(AppState::MainMenu);
20        }
21    }
22}
